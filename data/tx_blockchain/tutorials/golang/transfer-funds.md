# Transfer Funds with Golang

This document provides instructions and examples on how to use our `pkg/client` package to broadcast transactions and query the TX Blockchain using the Go programming language.

## Overview

| Component | Purpose |
|-----------|---------|
| `pkg/client` | TX Blockchain Go client library |
| `txd` | Core binary (used as dependency) |
| `gRPC` | Communication protocol with node |
| `Keyring` | Secure key management |

## Prerequisites

- [ ] Go 1.21+ installed (`go version`)
- [ ] `txd` module available (`go get github.com/tokenize-x/tx-chain/v3`)
- [ ] Funded testnet account (from faucet)

Step 5: Prepare Client Context and Tx Factory
go
func setupClient() (client.Context, client.Factory, *grpc.ClientConn) {
    // Setup modules
    modules := module.NewBasicManager(
        auth.AppModuleBasic{},
    )

    // Create encoding config
    encodingConfig := coreumconfig.NewEncodingConfig(modules)

    // Get gRPC codec provider
    pc, ok := encodingConfig.Codec.(codec.GRPCCodecProvider)
    if !ok {
        panic("failed to cast codec to codec.GRPCCodecProvider")
    }

    // Connect to gRPC node
    // For production: use TLS
    // For local/test: use grpc.WithTransportCredentials(insecure.NewCredentials())
    grpcClient, err := grpc.Dial(
        nodeAddress,
        grpc.WithDefaultCallOptions(grpc.ForceCodec(pc.GRPCCodec())),
        grpc.WithTransportCredentials(credentials.NewTLS(&tls.Config{MinVersion: tls.VersionTLS12})),
    )
    if err != nil {
        panic(err)
    }

    // Create client context
    clientCtx := client.NewContext(client.DefaultContextConfig(), modules).
        WithChainID(string(chainID)).
        WithGRPCClient(grpcClient).
        WithKeyring(keyring.NewInMemory(encodingConfig.Codec)).
        WithBroadcastMode(flags.BroadcastSync)

    // Create tx factory
    txFactory := client.Factory{}.
        WithKeybase(clientCtx.Keyring()).
        WithChainID(clientCtx.ChainID()).
        WithTxConfig(clientCtx.TxConfig()).
        WithSimulateAndExecute(true)

    return clientCtx, txFactory, grpcClient
}

func importAccount(clientCtx client.Context, mnemonic string) (keyring.Info, error) {
    // Import account from mnemonic
    senderInfo, err := clientCtx.Keyring().NewAccount(
        "sender",           // Key name
        mnemonic,           // Mnemonic phrase
        "",                 // BIP39 passphrase (empty if none)
        sdk.GetConfig().GetFullBIP44Path(), // HD path
        hd.Secp256k1,       // Signing algorithm
    )
    if err != nil {
        return nil, err
    }
    
    return senderInfo, nil
}

Step 7: Get Account Address

func getAddress(senderInfo keyring.Info) (sdk.AccAddress, error) {
    // Get address from key info
    senderAddress, err := senderInfo.GetAddress()
    if err != nil {
        return nil, err
    }
    
    fmt.Printf("Sender Address: %s\n", senderAddress.String())
    return senderAddress, nil
}

func validateAddress(address sdk.AccAddress) error {
    // Validate address format
    _, err := sdk.AccAddressFromBech32(address.String())
    if err != nil {
        return fmt.Errorf("invalid address: %w", err)
    }
    return nil
}
Step 8: Broadcast Transaction
go
func sendTokens(
    ctx context.Context,
    clientCtx client.Context,
    txFactory client.Factory,
    fromAddress sdk.AccAddress,
    toAddress string,
    amount int64,
    denom string,
) (*client.BroadcastTxResult, error) {
    
    // Create message
    msg := &banktypes.MsgSend{
        FromAddress: fromAddress.String(),
        ToAddress:   toAddress,
        Amount:      sdk.NewCoins(sdk.NewInt64Coin(denom, amount)),
    }
    
    // Broadcast transaction
    result, err := client.BroadcastTx(
        ctx,
        clientCtx.WithFromAddress(fromAddress),
        txFactory,
        msg,
    )
    if err != nil {
        return nil, err
    }
    
    fmt.Printf("Tx hash: %s\n", result.TxHash)
    return result, nil
}
Step 9: Query Balance
go
func queryBalance(
    ctx context.Context,
    clientCtx client.Context,
    address string,
) (sdk.Coins, error) {
    
    // Create bank query client
    bankClient := banktypes.NewQueryClient(clientCtx)
    
    // Query all balances
    balances, err := bankClient.AllBalances(ctx, &banktypes.QueryAllBalancesRequest{
        Address: address,
    })
    if err != nil {
        return nil, err
    }
    
    fmt.Printf("Balances: %s\n", balances.Balances)
    return balances.Balances, nil
}
Step 10: Complete Main Function
go
func main() {
    // Setup Cosmos SDK config
    setupConfig()
    
    // Setup client
    clientCtx, txFactory, grpcClient := setupClient()
    defer grpcClient.Close()
    
    // Import account from mnemonic
    senderInfo, err := importAccount(clientCtx, senderMnemonic)
    if err != nil {
        log.Fatalf("Failed to import account: %v", err)
    }
    
    // Get and validate address
    senderAddress, err := getAddress(senderInfo)
    if err != nil {
        log.Fatalf("Failed to get address: %v", err)
    }
    
    if err := validateAddress(senderAddress); err != nil {
        log.Fatalf("Address validation failed: %v", err)
    }
    
    ctx := context.Background()
    
    // Query initial balance
    fmt.Println("=== Initial Balance ===")
    initialBalance, err := queryBalance(ctx, clientCtx, senderAddress.String())
    if err != nil {
        log.Printf("Warning: Could not query balance: %v", err)
    }
    
    // Send tokens
    fmt.Println("\n=== Sending Tokens ===")
    amount := int64(9_000_000) // 9 utestcore
    result, err := sendTokens(
        ctx,
        clientCtx,
        txFactory,
        senderAddress,
        recipientAddress,
        amount,
        denom,
    )
    if err != nil {
        log.Fatalf("Failed to send tokens: %v", err)
    }
    
    fmt.Printf("Transaction successful! Hash: %s\n", result.TxHash)
    
    // Query updated balance
    fmt.Println("\n=== Updated Balance ===")
    newBalance, err := queryBalance(ctx, clientCtx, senderAddress.String())
    if err != nil {
        log.Printf("Warning: Could not query balance: %v", err)
    }
    
    // Query recipient balance
    fmt.Println("\n=== Recipient Balance ===")
    recipientBalance, err := queryBalance(ctx, clientCtx, recipientAddress)
    if err != nil {
        log.Printf("Warning: Could not query recipient balance: %v", err)
    }
    
    _ = initialBalance
    _ = newBalance
    _ = recipientBalance
}
