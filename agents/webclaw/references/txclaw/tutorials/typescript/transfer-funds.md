# Transfer Funds with TypeScript

This document provides instructions and examples on how to use the CosmJS package to broadcast transactions and query the TX Blockchain using TypeScript.

> **Note**: The tutorial "Using CosmJS" includes the part with funds transfer. This document expands on that with additional examples and detailed explanations.

## What Is CosmJS?

CosmJS is a JavaScript library designed for building applications that can interact with blockchains built with the Cosmos SDK. The Cosmos SDK is a framework for building blockchain applications in Go, and it powers a wide range of blockchains in the Cosmos Network.

CosmJS provides developers with:
- Easy connection to Cosmos SDK-based blockchains
- Transaction creation and broadcasting
- Blockchain data querying
- Wallet management
- Cryptographic functions
- Communication via REST and RPC endpoints

Step 4: Initialize Clients
typescript
import { Tendermint34Client } from "@cosmjs/tendermint-rpc";
import { QueryClient } from "@cosmjs/stargate/build/queryclient/queryclient";
import { createProtobufRpcClient } from "@cosmjs/stargate";
import { setupStakingExtension } from "@cosmjs/stargate/build/modules";
import { QueryClientImpl as FeemodelQueryClient } from "../coreum-ts/coreum/feemodel/v1/query";

async function initializeClients(rpcEndpoint: string) {
    // Connect to Tendermint RPC
    const tendermintClient = await Tendermint34Client.connect(rpcEndpoint);
    
    // Create query client
    const queryClient = new QueryClient(tendermintClient);
    
    // Create protobuf RPC client
    const rpcClient = createProtobufRpcClient(queryClient);
    
    // Initialize custom query clients
    const feemodelQueryClient = new FeemodelQueryClient(rpcClient);
    const stakingExtension = setupStakingExtension(queryClient);
    
    return {
        tendermintClient,
        queryClient,
        rpcClient,
        feemodelQueryClient,
        stakingExtension,
    };
}

Step 7: Get Gas Price with Multiplier
TX Blockchain uses a dynamic fee model. The gas price should be fetched and multiplied based on expected wait time:

typescript
import { GasPrice, decodeCosmosSdkDecFromProto } from "@cosmjs/stargate";

export async function getGasPriceWithMultiplier(
    feemodelQueryClient: FeemodelQueryClient,
    multiplier: number = 1.1
): Promise<GasPrice> {
    // Query current minimum gas price
    const minGasPriceRes = await feemodelQueryClient.MinGasPrice({});
    
    // Decode the amount
    const minGasPrice = decodeCosmosSdkDecFromProto(
        minGasPriceRes.minGasPrice?.amount || ""
    );
    
    // Apply multiplier
    let gasPrice = minGasPrice.toFloatApproximation() * multiplier;
    
    return GasPrice.fromString(
        `${gasPrice}${minGasPriceRes.minGasPrice?.denom || ""}`
    );
}
Multiplier Recommendations
Wait Time	Multiplier	Use Case
Immediate	1.1	Send right after query
< 30 seconds	1.3	Moderate delay
< 60 seconds	1.5	Significant delay
> 60 seconds	Initial price	Query again
Step 8: Send Tokens (Bank Transfer)
typescript
import { SigningStargateClient, calculateFee } from "@cosmjs/stargate";
import { isDeliverTxSuccess } from "@cosmjs/stargate/build/stargateclient";
import { MsgSend } from "cosmjs-types/cosmos/bank/v1beta1/tx";
import { MsgSendEncodeObject } from "@cosmjs/stargate/build/modules";

async function sendTokens(
    rpcEndpoint: string,
    senderWallet: DirectSecp256k1HdWallet,
    senderAddress: string,
    recipientAddress: string,
    amount: string,
    denom: string,
    feemodelQueryClient: FeemodelQueryClient,
    registry: Registry
) {
    // Connect to client
    const client = await SigningStargateClient.connectWithSigner(
        rpcEndpoint,
        senderWallet,
        { registry }
    );
    
    // Check sender balance
    const senderBalance = await client.getBalance(senderAddress, denom);
    console.log(`Sender balance: ${senderBalance.amount}${denom}`);
    
    // Create transfer message
    const msgSend: MsgSendEncodeObject = {
        typeUrl: "/cosmos.bank.v1beta1.MsgSend",
        value: MsgSend.fromPartial({
            fromAddress: senderAddress,
            toAddress: recipientAddress,
            amount: [{
                denom: denom,
                amount: amount,
            }],
        }),
    };
    
    console.log(`Sending ${amount}${denom} from ${senderAddress} to ${recipientAddress}`);
    
    // Get gas price and simulate
    const gasPrice = await getGasPriceWithMultiplier(feemodelQueryClient);
    const gasEstimate = await client.simulate(senderAddress, [msgSend], "");
    console.log(`Estimated gas: ${gasEstimate}, Gas price: ${gasPrice.toString()}`);
    
    // Calculate fee
    const fee = calculateFee(gasEstimate, gasPrice);
    
    // Sign and broadcast
    const result = await client.signAndBroadcast(
        senderAddress,
        [msgSend],
        fee
    );
    
    if (!isDeliverTxSuccess(result)) {
        throw new Error(`Transaction failed: ${result.rawLog}`);
    }
    
    console.log(`✅ Transaction successful! Tx hash: ${result.transactionHash}`);
    
    // Check recipient balance
    const recipientBalance = await client.getBalance(recipientAddress, denom);
    console.log(`Recipient balance: ${recipientBalance.amount}${denom}`);
    
    return result;
}

import { MsgDelegate } from "cosmjs-types/cosmos/staking/v1beta1/tx";
import { MsgDelegateEncodeObject } from "@cosmjs/stargate/build/modules";

async function delegateTokens(
    rpcEndpoint: string,
    wallet: DirectSecp256k1HdWallet,
    delegatorAddress: string,
    validatorAddress: string,
    amount: string,
    denom: string,
    feemodelQueryClient: FeemodelQueryClient,
    stakingExtension: any,
    registry: Registry
) {
    const client = await SigningStargateClient.connectWithSigner(
        rpcEndpoint,
        wallet,
        { registry }
    );
    
    // Create delegation message
    const msgDelegate: MsgDelegateEncodeObject = {
        typeUrl: "/cosmos.staking.v1beta1.MsgDelegate",
        value: MsgDelegate.fromPartial({
            delegatorAddress: delegatorAddress,
            validatorAddress: validatorAddress,
            amount: {
                denom: denom,
                amount: amount,
            },
        }),
    };
    
    console.log(`Delegating ${amount}${denom} to ${validatorAddress}`);
    
    // Get gas price
    const gasPrice = await getGasPriceWithMultiplier(feemodelQueryClient);
    const gasEstimate = await client.simulate(delegatorAddress, [msgDelegate], "");
    const fee = calculateFee(gasEstimate, gasPrice);
    
    // Sign and broadcast
    const result = await client.signAndBroadcast(
        delegatorAddress,
        [msgDelegate],
        fee
    );
    
    if (!isDeliverTxSuccess(result)) {
        throw new Error(`Delegation failed: ${result.rawLog}`);
    }
    
    console.log(`✅ Delegation successful! Tx hash: ${result.transactionHash}`);
    
    return result;
}

// Get bonded validators
async function getBondedValidators(stakingExtension: any) {
    const bondedValidators = await stakingExtension.staking.validators("BOND_STATUS_BONDED");
    
    if (bondedValidators.validators.length === 0) {
        throw new Error("No bonded validators found");
    }
    
    const validator = bondedValidators.validators[0];
    console.log(`Using validator: ${validator.description?.moniker || validator.operatorAddress}`);
    
    return validator.operatorAddress;

        senderWallet,

        prefix: TX_ACCOUNT_PREFIX,
        hdPaths: [stringToPath(TX_HD_PATH)],

    console.log("\n=== Sending Tokens ===");

    const msgSend: MsgSendEncodeObject = {
        typeUrl: "/cosmos.bank.v1beta1.MsgSend",
        value: MsgSend.fromPartial({
            fromAddress: sender.address,
            toAddress: recipient.address,
            amount: [{
                denom: TX_DENOM,

        sendFee

    if (!isDeliverTxSuccess(sendResult)) {

Error Handling
typescript
import { isDeliverTxFailure } from "@cosmjs/stargate";

function handleTxError(result: any) {
    if (isDeliverTxFailure(result)) {
        switch (result.code) {
            case 11:
                console.error("Insufficient funds for transaction");
                break;
            case 13:
                console.error("Invalid address format");
                break;
            case 19:
                console.error("Gas limit exceeded");
                break;
            default:
                console.error(`Transaction failed with code ${result.code}: ${result.rawLog}`);
        }
        return false;
    }
    return true;
}
