# Record Service

The Record Service provides API interfaces to retrieve users' records, such as tax reports and financial statements.

## Overview

The Record Service allows authenticated users to access their financial documents, tax reports, account statements, and other official records generated by the platform.

## API Endpoints

All endpoints are **authenticated** and require:
- Valid Firebase JWT token in `Authorization` header (format: `Bearer: eyJhb....`)
- Organization ID in `OrganizationID` header
- Network in `Network` header (mainnet, testnet, devnet)

| Method | Endpoint | Description |
|--------|----------|-------------|
| GET | `/api/record/get` | Retrieve a single record |
| GET | `/api/record/list` | Retrieve a list of records |

## GET /api/record/get

Retrieve a single record for a user. All query parameters are required.

### Query Parameters

| Parameter | Type | Required | Description |
|-----------|------|----------|-------------|
| type | int | Yes | Type of record (e.g., 1 for tax report, 2 for statement) |
| period_id | string | Yes | Period identifier (e.g., "2023", "2023-Q1", "2024-05") |

class RecordServiceClient {
  constructor(baseUrl, token, organizationId, network = 'mainnet') {
    this.baseUrl = baseUrl;
    this.token = token;
    this.organizationId = organizationId;
    this.network = network;
  }

  async getRecord(type, periodId) {
    const url = new URL(`${this.baseUrl}/api/record/get`);
    url.searchParams.append('type', type);
    url.searchParams.append('period_id', periodId);

    const response = await fetch(url.toString(), {
      headers: this._getHeaders()
    });

    if (!response.ok) {
      throw new Error(`Failed to fetch record: ${response.status}`);
    }

    return response.json();
  }

  async listRecords(filter = null) {
    const url = new URL(`${this.baseUrl}/api/record/list`);
    
    if (filter) {
      const filterBase64 = btoa(JSON.stringify(filter));
      url.searchParams.append('filter', filterBase64);
    }

    const response = await fetch(url.toString(), {
      headers: this._getHeaders()
    });

    if (!response.ok) {
      throw new Error(`Failed to list records: ${response.status}`);
    }

    return response.json();
  }

  async getTaxReport(year) {
    return this.getRecord(1, year.toString());
  }

  async getQuarterlyTaxReport(year, quarter) {
    return this.getRecord(1, `${year}-Q${quarter}`);
  }

  async getMonthlyStatement(year, month) {
    const periodId = `${year}-${String(month).padStart(2, '0')}`;
    return this.getRecord(2, periodId);
  }

  async getAllTaxReports() {
    return this.listRecords({ Type: 1 });
  }

  async getAllStatements() {
    return this.listRecords({ Type: 2 });
  }

  _getHeaders() {
    return {
      'Authorization': `Bearer: ${this.token}`,
      'OrganizationID': this.organizationId,
      'Network': this.network
    };
  }
}

function RecordViewer({ userId }) {
  const [records, setRecords] = useState([]);
  const [loading, setLoading] = useState(true);
  const [selectedYear, setSelectedYear] = useState(new Date().getFullYear());

  const client = new RecordServiceClient(
    'https://api.sologenic.org',
    localStorage.getItem('token'),
    localStorage.getItem('orgId'),
    'mainnet'
  );

  useEffect(() => {
    async function loadRecords() {
      try {
        const taxReport = await client.getTaxReport(selectedYear);
        const statements = await client.getAllStatements();
        setRecords([taxReport.Record, ...statements.Records.map(r => r.Record)]);
      } catch (err) {
        console.error('Failed to load records:', err);
      } finally {
        setLoading(false);
      }
    }
    loadRecords();
  }, [selectedYear]);

  if (loading) return <div>Loading records...</div>;

  return (
    <div className="record-viewer">
      <h2>Financial Records</h2>
      
      <div className="year-selector">
        <label>Tax Year:</label>
        <select value={selectedYear} onChange={(e) => setSelectedYear(e.target.value)}>
          <option value="2023">2023</option>
          <option value="2024">2024</option>
          <option value="2025">2025</option>
        </select>
      </div>

      <div className="records-list">
        {records.map((record, idx) => (
          <div key={idx} className="record-card">
            <h3>{record.Title}</h3>
            <p>{record.Description}</p>

  );
}

import requests
import base64
import json
from typing import Optional, Dict, Any, List
from dataclasses import dataclass
from datetime import datetime

@dataclass
class Record:
    UserID: str
    OrganizationID: str
    Title: str
    Description: str
    Type: int
    Period: int
    PeriodIdentifier: str
    PeriodFrom: int
    PeriodTo: int
    FileReference: str
    FileExtension: str
    FileName: str

class RecordServiceClient:
    def __init__(self, base_url: str, token: str, organization_id: str, network: str = 'mainnet'):
        self.base_url = base_url
        self.token = token
        self.organization_id = organization_id
        self.network = network
    
    def get_record(self, record_type: int, period_id: str) -> Dict[str, Any]:
        """Get a single record by type and period"""
        response = requests.get(
            f"{self.base_url}/api/record/get",
            params={'type': record_type, 'period_id': period_id},
            headers=self._get_headers()
        )
        response.raise_for_status()
        return response.json()
    
    def list_records(self, filter_obj: Optional[Dict] = None) -> List[Dict]:
        """List all records with optional filter"""
        params = {}
        if filter_obj:
            filter_json = json.dumps(filter_obj)
            filter_base64 = base64.b64encode(filter_json.encode()).decode()
            params['filter'] = filter_base64
        
        response = requests.get(
            f"{self.base_url}/api/record/list",
            params=params,
            headers=self._get_headers()
        )
        response.raise_for_status()
        return response.json().get('Records', [])
    
    def get_tax_report(self, year: int) -> Dict:
        """Get annual tax report for a specific year"""
        return self.get_record(1, str(year))
    
    def get_quarterly_tax_report(self, year: int, quarter: int) -> Dict:
        """Get quarterly tax report"""
        return self.get_record(1, f"{year}-Q{quarter}")
    
    def get_monthly_statement(self, year: int, month: int) -> Dict:
        """Get monthly account statement"""
        period_id = f"{year}-{month:02d}"
        return self.get_record(2, period_id)
    
    def get_all_tax_reports(self) -> List[Dict]:
        """Get all tax reports"""
        return self.list_records({'Type': 1})
    
    def get_all_statements(self) -> List[Dict]:
        """Get all account statements"""
        return self.list_records({'Type': 2})
    
    def download_record_file(self, record: Dict, output_path: str) -> None:
        """Download the actual file for a record"""
        file_info = record['Record']['File']
        download_url = f"{self.base_url}/api/record/download?ref={file_info['Reference']}"
        
        response = requests.get(
            download_url,
            headers=self._get_headers(),
            stream=True
        )
        response.raise_for_status()
        
        filename = f"{file_info['Name']}.{file_info['Extension']}"
        filepath = f"{output_path}/{filename}"
        
        with open(filepath, 'wb') as f:
            for chunk in response.iter_content(chunk_size=8192):
                f.write(chunk)
        
        return filepath
    
    def _get_headers(self) -> Dict[str, str]:
        return {
            'Authorization': f'Bearer: {self.token}',
            'OrganizationID': self.organization_id,
            'Network': self.network
        }

package main

import (
    "encoding/base64"
    "encoding/json"
    "fmt"
    "io"
    "net/http"
    "net/url"
    "time"
)

type RecordServiceClient struct {
    baseURL        string
    token          string
    organizationID string
    network        string
    httpClient     *http.Client
}

type Record struct {
    UserID         string    `json:"UserID"`
    OrganizationID string    `json:"OrganizationID"`
    Title          string    `json:"Title"`
    Description    string    `json:"Description"`
    Type           int       `json:"Type"`
    Period         int       `json:"Period"`
    PeriodIdentifier string  `json:"PeriodIdentifier"`
    PeriodFrom     Timestamp `json:"PeriodFrom"`
    PeriodTo       Timestamp `json:"PeriodTo"`
    File           FileInfo  `json:"File"`
}

type FileInfo struct {
    Reference string `json:"Reference"`
    Extension string `json:"Extension"`
    Name      string `json:"Name"`
}

type Timestamp struct {
    Seconds int64 `json:"seconds"`
    Nanos   int32 `json:"nanos"`
}

type RecordResponse struct {
    Record   Record   `json:"Record"`
    MetaData Metadata `json:"MetaData"`
    Audit    Audit    `json:"Audit"`
}

type Metadata struct {
    Network   int       `json:"Network"`
    CreatedAt Timestamp `json:"CreatedAt"`
    UpdatedAt Timestamp `json:"UpdatedAt"`
}

type Audit struct {
    ChangedBy string    `json:"ChangedBy"`
    ChangedAt Timestamp `json:"ChangedAt"`
}

type ListRecordsResponse struct {
    Records []RecordResponse `json:"Records"`
}

func NewRecordServiceClient(baseURL, token, organizationID, network string) *RecordServiceClient {
    return &RecordServiceClient{
        baseURL:        baseURL,
        token:          token,
        organizationID: organizationID,
        network:        network,
        httpClient:     &http.Client{Timeout: 30 * time.Second},
    }
}

func (c *RecordServiceClient) GetRecord(recordType int, periodID string) (*RecordResponse, error) {
    params := url.Values{}
    params.Add("type", fmt.Sprintf("%d", recordType))
    params.Add("period_id", periodID)
    
    reqURL := fmt.Sprintf("%s/api/record/get?%s", c.baseURL, params.Encode())
    
    req, err := http.NewRequest("GET", reqURL, nil)
    if err != nil {
        return nil, err
    }
    
    c.setHeaders(req)
    
    resp, err := c.httpClient.Do(req)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()
    
    body, err := io.ReadAll(resp.Body)
    if err != nil {
        return nil, err
    }
    
    var result RecordResponse
    if err := json.Unmarshal(body, &result); err != nil {
        return nil, err
    }
    
    return &result, nil
}

func (c *RecordServiceClient) ListRecords(filter map[string]interface{}) (*ListRecordsResponse, error) {
    reqURL := fmt.Sprintf("%s/api/record/list", c.baseURL)
    
    if filter != nil {
        filterJSON, err := json.Marshal(filter)
        if err != nil {
            return nil, err
        }
        filterBase64 := base64.URLEncoding.EncodeToString(filterJSON)
        reqURL = fmt.Sprintf("%s?filter=%s", reqURL, filterBase64)
    }
    
    req, err := http.NewRequest("GET", reqURL, nil)
    if err != nil {
        return nil, err
    }
    
    c.setHeaders(req)
    
    resp, err := c.httpClient.Do(req)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()
    
    body, err := io.ReadAll(resp.Body)
    if err != nil {
        return nil, err
    }
    
    var result ListRecordsResponse
    if err := json.Unmarshal(body, &result); err != nil {
        return nil, err
    }
    
    return &result, nil
}

func (c *RecordServiceClient) GetTaxReport(year int) (*RecordResponse, error) {
    return c.GetRecord(1, fmt.Sprintf("%d", year))
}

func (c *RecordServiceClient) GetMonthlyStatement(year, month int) (*RecordResponse, error) {
    periodID := fmt.Sprintf("%d-%02d", year, month)
    return c.GetRecord(2, periodID)
}

func (c *RecordServiceClient) GetAllTaxReports() (*ListRecordsResponse, error) {
    return c.ListRecords(map[string]interface{}{"Type": 1})
}

func (c *RecordServiceClient) setHeaders(req *http.Request) {
    req.Header.Set("Authorization", "Bearer: "+c.token)
    req.Header.Set("OrganizationID", c.organizationID)
    req.Header.Set("Network", c.network)
}

Troubleshooting

Filter not working	Invalid base64 encoding	Ensure proper base64 encoding of JSON
Best Practices
Security
Always include all required headers

Never expose tokens in client-side code

Validate organization access before requests

Use HTTPS in production

Performance
Cache frequently accessed records

Use filters to limit result sets

Implement pagination for large lists

Download files asynchronously

Data Management
Verify period identifiers before querying

Check record type values against documentation

Handle missing records gracefully

Log audit information for compliance
